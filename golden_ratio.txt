    h = a + b
<=> b = h - a

    a / b = golden ratio
<=> a = golden ratio * (h - a)
<=> a = golden ratio * h - goldren ratio * a
<=> (1 + golden ratio) * a = golden ratio * h
<=> a = (golden ratio * h) / (1 + golden ratio)
<=> a = h * (golden ratio / (1 + golden ratio))

golden ratio / (1 + golden ratio) ≅ 0.61803398875
